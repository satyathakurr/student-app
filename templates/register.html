@app.route("/register", methods=["GET", "POST"]) def register(): if request.method == "POST": username = request.form["username"] password = request.form["password"] # ✅ ADD VALIDATION HERE if not username or not password: flash("All fields are required!") return redirect("/register") try: conn = get_db() conn.execute( "INSERT INTO users (username, password) VALUES (?, ?)", (username, password) ) conn.commit() conn.close() flash("User registered!") return redirect("/login") except Exception as e: print(e) flash("Username already exists!") return render_template("register.html")
